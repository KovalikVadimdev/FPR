## Недописанная программа на Python
___ 

```Python
# coding: cp1251
## ============================================================
##  К лаб. работе № 3 по курсу "функциональное программирование"
## ============================================================
import random
import math
## ====================  Main program  =========================
Protocol_name = "Protocol_name"   ## задаётся имя своего протокола
file  = open(Protocol_name, "a")  ## Протокол открывается на запись
print >> file, "======== Начало протокола ",Protocol_name,"========"
## A = 12345
## print >> file, "A = ", A
## gg = int(random.random() * 255)
nn = 20        ##  сколько данных
## print >> file, "gg = ", gg, "nn = ", nn
print >> file, "nn = ", nn
Q = [0] * nn     ##   определение и начальное заполнение нулями массива исходных данных
print >> file, "Q = ", Q
for i in range(0, nn):   ##  генерация данных и помещение в массив
       Q[i] = int(random.random() * 255)
print >> file, "Q = ", Q
MaxQ = max(Q); MinQ = min(Q); print >> file, MaxQ, MinQ    ##   диапазон данных
## Дальше должно быть определение границ для каждой из  5  ячеек гистограммы,
## прогонка массива Q, сравнение с границами ячеек и добавление "единиц" в соответствующие ячейки.
## Дальше - 5-элементный массив гистограммы должен быть выведен в протокол (в числовой форме).
print >> file, "======== Конец протокола ",Protocol_name,"========"
file.close()
print "        Protocol is generated"
z=0
while z != "":
  print "======================================="
  z = input("     Press ENTER for exit ... ")
```
